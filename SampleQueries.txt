query GetShares ($first:Int, $last:Int, $industryId:Int) {
  shares(first:$first, last:$last, industryId: $industryId) {
    totalCount

    nodes {
      ... Share
    }
  }
}

query GetIndustries {
  industries {
    ... Industry
  }
}

fragment Share on Share {
  id,
  companyName,
  value,
  industry {
    ... Industry
  }
}

fragment Industry on Industry {
  id
  name
}
